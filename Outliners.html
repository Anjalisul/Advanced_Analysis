#to find the outliner 
def detect_outliner(data,column):
    q1=data[column].quantile(q=0.25)
    q3=data[column].quantile(q=0.75)
    iqr=q3-q1
    
    lim_iqr = 1.5*iqr

    upper_iqr = q3 + lim_iqr
    lower_iqr = q1 - lim_iqr
    
    outliers_data = data[(data[column] > upper_iqr) | (data[column] < lower_iqr)]  
    return outliers_data[column].to_list();
